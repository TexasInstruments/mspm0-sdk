488 __STATIC_INLINE void HAL_updatePWM(const HAL_PWM posPhase,
489  const HAL_PWM gndPhase,
490  const HAL_PWM opnPhase,
491  const uint16_t PWMDuty)
492 {
493  HAL_Timer_Instance *positivePhase = &PWMBase[posPhase];
494  HAL_Timer_Instance *groundPhase = &PWMBase[gndPhase];
495  HAL_Timer_Instance *openPhase = &PWMBase[opnPhase];
496 
497  /* get the timer load value */
498  uint16_t PWMPeriodC0 = DL_TimerG_getLoadValue(positivePhase->gptimer);
499 
500  uint16_t deadBand = PWMBase[posPhase].deadBand;
501 
502  /* Calculate the capture compare value */
503  uint16_t posPhaseC0_cmpValue = PWMPeriodC0 - (PWMPeriodC0 * PWMDuty)/100 + deadBand;
504  uint16_t posPhaseC1_cmpValue = PWMPeriodC0 - (PWMPeriodC0 * PWMDuty)/100 - deadBand;
505 
506  /* Set the capture compare value to positive phase */
507  DL_TimerG_setCaptureCompareValue(positivePhase->gptimer,
508  posPhaseC0_cmpValue,
509  positivePhase->ccIndex);
510  DL_TimerG_setCaptureCompareValue(positivePhase->gptimer,
511  posPhaseC1_cmpValue,
512  positivePhase->ccCompIndex);
513 
514  DL_Timer_overrideCCPOut(positivePhase->gptimer, DL_TIMER_FORCE_OUT_DISABLED,
515  0, positivePhase->ccIndex);
516  DL_Timer_overrideCCPOut(positivePhase->gptimer, DL_TIMER_FORCE_OUT_DISABLED,
517  0, positivePhase->ccCompIndex);
518 
519  DL_Timer_overrideCCPOut(groundPhase->gptimer, DL_TIMER_FORCE_OUT_LOW,
520  0, groundPhase->ccIndex);
521  DL_Timer_overrideCCPOut(groundPhase->gptimer, DL_TIMER_FORCE_OUT_HIGH,
522  0, groundPhase->ccCompIndex);
523 
524  DL_Timer_overrideCCPOut(openPhase->gptimer, DL_TIMER_FORCE_OUT_LOW,
525  0, openPhase->ccIndex);
526  DL_Timer_overrideCCPOut(openPhase->gptimer, DL_TIMER_FORCE_OUT_LOW,
527  0, openPhase->ccCompIndex);
528 }
