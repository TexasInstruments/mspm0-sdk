551 __STATIC_INLINE void DRV8316_setnSleep(
552  DRV8316_Instance *drvHandle, DRV8316_DRV_NSLEEP_STAT value)
553 {
554  if(value == DRV8316_DRV_NSLEEP_AWAKE)
555  {
556  HAL_writeGPIOPin(drvHandle->nsleep, HAL_GPIO_PIN_HIGH);
557  }
558  else if(value == DRV8316_DRV_NSLEEP_SLEEP)
559  {
560  HAL_writeGPIOPin(drvHandle->nsleep, HAL_GPIO_PIN_LOW);
561  }
562  else
563  {
564  /* This is expected to be empty */
565  }
566  /* tWake - Turnon time = 1ms
567  * tSleep - Turnoff time = 1ms
568  * So not matter what you set sleep pin to just wait 1 ms
569  */
570  HAL_delayMicroSeconds(DRV8316_MAXIMUM_WAKEUP_TIME);
571 }
