26 /*!****************************************************************************
27  * @file hid_host.h
28  * @brief HID (Human Interface Device) Host Driver implementation
29  *
30  * @anchor tinyusb_class_hid_host
31  * # Overview
32  *
33  * This file implements the HID host driver for TinyUSB stack.
34  * It handles the USB HID protocol including:
35  * - Device enumeration and configuration
36  * - Report descriptor parsing
37  * - Input/Output/Feature report transfers
38  * - Protocol mode selection (Boot/Report)
39  * - Idle rate management
40  *
41  * The HID implementation supports:
42  * - Multiple HID interfaces
43  * - Various HID device types (keyboard, mouse, gamepad, etc.)
44  * - Boot protocol for keyboards and mice
45  * - Report protocol with descriptor parsing
46  * - Configurable report buffer sizes
47  * - Synchronous and asynchronous report transfers
48  *
49  * <hr>
50  ******************************************************************************/
