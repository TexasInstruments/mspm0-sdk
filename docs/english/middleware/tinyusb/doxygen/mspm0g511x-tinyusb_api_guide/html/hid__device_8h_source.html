26 /*!****************************************************************************
27  * @file hid_device.h
28  * @brief HID (Human Interface Device) Device Driver implementation
29  *
30  * @anchor tinyusb_class_hid_device
31  * # Overview
32  *
33  * This file implements the HID device driver for TinyUSB stack.
34  * It handles the USB HID protocol including:
35  * - Report descriptor definition and parsing
36  * - Input/Output/Feature report transfers
37  * - Protocol mode selection (Boot/Report)
38  * - Idle rate management
39  *
40  * The HID implementation supports:
41  * - Multiple HID interfaces
42  * - Various HID device types (keyboard, mouse, gamepad, etc.)
43  * - Boot protocol for keyboards and mice
44  * - Report protocol with descriptor templates
45  * - Configurable report buffer sizes
46  * - Synchronous and asynchronous report transfers
47  *
48  * <hr>
49  ******************************************************************************/
