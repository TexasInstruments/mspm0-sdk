26 /*!****************************************************************************
27  * @file cdc_host.h
28  * @brief CDC (Communication Device Class) Host Driver implementation
29  *
30  * @details This file implements the CDC host driver for TinyUSB stack.
31  * It handles the USB CDC protocol including:
32  * - Device enumeration and configuration
33  * - Serial data transfer between USB host and device
34  * - Control requests handling (line coding, control signals)
35  * - Buffer management for data transmission and reception
36  *
37  * The CDC host implementation supports:
38  * - Multiple CDC interfaces
39  * - Line coding configuration (baud rate, data bits, parity, stop bits)
40  * - Line state signaling (DTR/RTS)
41  * - Data transfer with connected CDC devices
42  *
43  * <hr>
44  ******************************************************************************/
