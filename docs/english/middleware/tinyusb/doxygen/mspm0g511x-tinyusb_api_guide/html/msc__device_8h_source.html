26 /*!****************************************************************************
27  * @file msc_device.h
28  * @brief MSC (Mass Storage Class) Device Driver implementation
29  *
30  * @anchor tinyusb_class_msc_device
31  * # Overview
32  *
33  * This file implements the MSC device driver for TinyUSB stack.
34  * It handles the USB MSC protocol including:
35  * - Device enumeration and configuration
36  * - SCSI command processing (read, write, inquiry, etc.)
37  * - Block data transfer between USB host and device storage
38  * - Multiple Logical Unit Number (LUN) support
39  *
40  * The MSC implementation supports:
41  * - Multiple storage interfaces
42  * - Configurable buffer sizes via CFG_TUD_MSC_EP_BUFSIZE
43  * - Read-only or read-write storage options
44  * - Standard SCSI commands and custom command handling
45  * - Ejection and unit ready status reporting
46  *
47  * <hr>
48  ******************************************************************************/
