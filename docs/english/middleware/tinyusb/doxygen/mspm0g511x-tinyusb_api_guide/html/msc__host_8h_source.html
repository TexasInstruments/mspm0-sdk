26 /*!****************************************************************************
27  * @file msc_host.h
28  * @brief MSC (Mass Storage Class) Host Driver implementation
29  *
30  * @anchor tinyusb_class_msc_host
31  * # Overview
32  *
33  * This file implements the MSC host driver for TinyUSB stack.
34  * It handles the USB MSC protocol including:
35  * - Device enumeration and configuration
36  * - SCSI command processing (read, write, inquiry, etc.)
37  * - Block data transfer between USB host and device storage
38  * - Multiple Logical Unit Number (LUN) support
39  *
40  * The MSC implementation supports:
41  * - Multiple storage interfaces
42  * - Configurable maximum LUN via CFG_TUH_MSC_MAXLUN
43  * - Standard SCSI commands and responses
44  * - Block read/write operations
45  *
46  * <hr>
47  ******************************************************************************/
