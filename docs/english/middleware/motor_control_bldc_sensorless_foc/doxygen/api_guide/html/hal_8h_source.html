442 __STATIC_INLINE DL_ADC12_IIDX HAL_processADCIRQ(ADC12_Regs *adcInst)
443 {
444  DL_ADC12_IIDX pendIrq = DL_ADC12_getPendingInterrupt(adcInst);
445  for(uint32_t x = 0;x < HAL_ADC_CHANNEL_MAX;x++)
446  {
447  if(adcInst == halAdc[x].inst && pendIrq == halAdc[x].endOfSeq)
448  {
449  halAdc[x].value = (uint16_t)HAL_getADCData(x);
450  }
451  }
452  return pendIrq;
453 }
