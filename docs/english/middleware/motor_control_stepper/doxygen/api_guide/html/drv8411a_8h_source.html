390 __STATIC_INLINE void DRV8411A_setPWMDrive(
391  HAL_GPIO_OUT_PIN in1, HAL_GPIO_OUT_PIN in2,
392  STEPPER_DRIVE drive, DRV8411A_DECAY decayMode)
393 {
394  HAL_GPIO_VALUE in1GpioVal = HAL_GPIO_VALUE_LOW;
395  HAL_GPIO_VALUE in2GpioVal = HAL_GPIO_VALUE_LOW;
396  switch (drive)
397  {
398  case STEPPER_DRIVE_DISABLE:
399  in1GpioVal = HAL_GPIO_VALUE_LOW;
400  in2GpioVal = HAL_GPIO_VALUE_LOW;
401  break;
402  case STEPPER_DRIVE_DECAY:
403  if(decayMode == DRV8411A_DECAY_SLOW)
404  {
405  in1GpioVal = HAL_GPIO_VALUE_HIGH;
406  in2GpioVal = HAL_GPIO_VALUE_HIGH;
407  }
408  else
409  {
410  in1GpioVal = HAL_GPIO_VALUE_LOW;
411  in2GpioVal = HAL_GPIO_VALUE_LOW;
412  }
413  break;
414  case STEPPER_DRIVE_FORWARD:
415  in1GpioVal = HAL_GPIO_VALUE_HIGH;
416  in2GpioVal = HAL_GPIO_VALUE_LOW;
417  break;
418  case STEPPER_DRIVE_REVERSE:
419  in1GpioVal = HAL_GPIO_VALUE_LOW;
420  in2GpioVal = HAL_GPIO_VALUE_HIGH;
421  break;
422  default:
423  /* This is expected to be empty */
424  break;
425  }
426  HAL_setGPIOVal(in1, in1GpioVal);
427  HAL_setGPIOVal(in2, in2GpioVal);
428 }

444 
445  DRV8411A_setPWMDrive(handle->bin1, handle->bin2,
446  indexer->IbDrive, handle->decayMode);
