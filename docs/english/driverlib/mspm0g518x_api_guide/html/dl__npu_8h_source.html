32 /*!****************************************************************************
33  * @file dl_npu.h
34  * @brief Neural Processing Unit (NPU) Driver Library
35  * @defgroup Neural Processing Unit (NPU)
36  *
37  * @anchor ti_dl_dl_npu_Overview
38  * # Overview
39  *
40  * The Neural Processing Unit (NPU) Driver Library provides low level
41  * NPU drivers for power control, interrupt management, and access
42  * to NPU control registers as well as NPU memory.
43  * The power control and interrupt management registers are expected to be
44  * called as a part of the user application, but the NPU control register
45  * and NPU memory address accessor functions are only intended to be used by
46  * the NPU library.
47  *
48  * <hr>
49  ******************************************************************************
50  */
