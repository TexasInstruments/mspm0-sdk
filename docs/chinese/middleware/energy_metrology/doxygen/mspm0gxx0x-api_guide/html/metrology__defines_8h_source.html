619 __STATIC_INLINE int64_t int64mpy(int32_t Input1, int32_t Input2)
620 {
621  int32_t uLow,vLow;
622  int32_t uHigh,vHigh;
623  int64_t result;
624 
625  /* Implement MULHS, high side of 64-bit 32bitx32bit multiplication */
626  uLow = Input1 & 0xFFFF;
627  vLow = Input2 & 0xFFFF;
628 
629  uHigh = Input1 >> 16;
630  vHigh = Input2 >> 16;
631 
632  result = (uLow*vLow);
633  result += (int64_t)(uHigh*vLow + uLow*vHigh) <<16 ;
634  result += (int64_t)(uHigh*vHigh) << 32;
635 
636  return (int64_t)result;
637 }
