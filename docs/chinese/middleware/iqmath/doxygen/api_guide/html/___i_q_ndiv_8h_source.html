39 __STATIC_INLINE int_fast32_t __IQNdiv(int_fast32_t iqNInput1, int_fast32_t iqNInput2, const uint8_t type, const int8_t q_value)
40 {
41  uint8_t ui8Index, ui8Sign = 0;
42  uint_fast32_t ui32Temp;
43  uint_fast32_t uiq30Guess;
44  uint_fast32_t uiqNInput1;
45  uint_fast32_t uiqNInput2;
46  uint_fast32_t uiqNResult;
47  uint_fast64_t uiiqNInput1;
48  uint_fast16_t ui16IntState;
49  uint_fast16_t ui16MPYState;
50 
51  if (type == TYPE_DEFAULT) {
52  /* save sign of denominator */
53  if (iqNInput2 <= 0) {
54  /* check for divide by zero */
55  if (iqNInput2 == 0) {
56  return INT32_MAX;
57  }
58  else {
59  ui8Sign = 1;
60  iqNInput2 = -iqNInput2;
61  }
62  }
63 
64  /* save sign of numerator */
65  if (iqNInput1 < 0) {
66  ui8Sign ^= 1;
67  iqNInput1 = -iqNInput1;
68  }
69  }
70  else {
71  /* Check for divide by zero */
72  if (iqNInput2 == 0) {
73  return INT32_MAX;
74  }
75  }
76 
77  /* Save input1 and input2 to unsigned IQN and IIQN (64-bit). */
78  uiiqNInput1 = (uint_fast64_t)iqNInput1;
79  uiqNInput2 = (uint_fast32_t)iqNInput2;
80 
81  /* Scale inputs so that 0.5 <= uiqNInput2 < 1.0. */
82  while (uiqNInput2 < 0x40000000) {
83  uiqNInput2 <<= 1;
84  uiiqNInput1 <<= 1;
85  }
86 
87  /*
88  * Shift input1 back from iq31 to iqN but scale by 2 since we multiply
89  * by result in iq30 format.
90  */
91  if (q_value < 31) {
92  uiiqNInput1 >>= (31 - q_value - 1);
93  }
94  else {
95  uiiqNInput1 <<= 1;
96  }
97 
98  /* Check for saturation. */
99  if (uiiqNInput1 >> 32) {
100  if (ui8Sign) {
101  return INT32_MIN;
102  }
103  else {
104  return INT32_MAX;
105  }
106  }
107  else {
108  uiqNInput1 = (uint_fast32_t)uiiqNInput1;
109  }
110 
111  /* use left most 7 bits as ui8Index into lookup table (range: 32-64) */
112  ui8Index = uiqNInput2 >> 24;
113  ui8Index -= 64;
114  uiq30Guess = (uint_fast32_t)_IQ6div_lookup[ui8Index] << 24;
115 
116  /*
117  * Mark the start of any multiplies. This will disable interrupts and set
118  * the multiplier to fractional mode. This is designed to reduce overhead
119  * of constantly switching states when using repeated multiplies (MSP430
120  * only).
121  */
122  __mpyf_start(&ui16IntState, &ui16MPYState);
123 
124  /* 1st iteration */
125  ui32Temp = __mpyf_ul(uiq30Guess, uiqNInput2);
126  ui32Temp = -((uint_fast32_t)ui32Temp - 0x80000000);
127  ui32Temp = ui32Temp << 1;
128  uiq30Guess = __mpyf_ul_reuse_arg1(uiq30Guess, ui32Temp);
129 
130  /* 2nd iteration */
131  ui32Temp = __mpyf_ul(uiq30Guess, uiqNInput2);
132  ui32Temp = -((uint_fast32_t)ui32Temp - 0x80000000);
133  ui32Temp = ui32Temp << 1;
134  uiq30Guess = __mpyf_ul_reuse_arg1(uiq30Guess, ui32Temp);
135 
136  /* 3rd iteration */
137  ui32Temp = __mpyf_ul(uiq30Guess, uiqNInput2);
138  ui32Temp = -((uint_fast32_t)ui32Temp - 0x80000000);
139  ui32Temp = ui32Temp << 1;
140  uiq30Guess = __mpyf_ul_reuse_arg1(uiq30Guess, ui32Temp);
141 
142  /* Multiply 1/uiqNInput2 and uiqNInput1. */
143  uiqNResult = __mpyf_ul(uiq30Guess, uiqNInput1);
144 
145  /*
146  * Mark the end of all multiplies. This restores MPY and interrupt states
147  * (MSP430 only).
148  */
149  __mpy_stop(&ui16IntState, &ui16MPYState);
150 
151  /* Saturate, add the sign and return. */
152  if (type == TYPE_DEFAULT) {
153  if (uiqNResult > INT32_MAX) {
154  if (ui8Sign) {
155  return INT32_MIN;
156  }
157  else {
158  return INT32_MAX;
159  }
160  }
161  else {
162  if (ui8Sign) {
163  return -(int_fast32_t)uiqNResult;
164  }
165  else {
166  return (int_fast32_t)uiqNResult;
167  }
168  }
169  }
170  else {
171  return uiqNResult;
172  }
173 }
