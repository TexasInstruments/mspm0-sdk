217 __STATIC_INLINE void drv8317_setnSleep(
218  drv8317_Instance *drvHandle, drv8317_DRV_NSLEEP_STAT value)
219 {
220  if(value == drv8317_DRV_NSLEEP_AWAKE)
221  {
222  HAL_writeGPIOPin(drvHandle->nsleep, HAL_GPIO_PIN_HIGH);
223  }
224  else if(value == drv8317_DRV_NSLEEP_SLEEP)
225  {
226  HAL_writeGPIOPin(drvHandle->nsleep, HAL_GPIO_PIN_LOW);
227  }
228  else
229  {
230  /* This is expected to be empty */
231  }
232  /* tWake - Turnon time = 1ms
233  * tSleep - Turnoff time = 1ms
234  * So not matter what you set sleep pin to just wait 1 ms
235  */
236  HAL_delayMicroSeconds(drv8317_MAXIMUM_WAKEUP_TIME);
237 }
