131 __STATIC_INLINE void PI_run(PI_Instance *handle)
132 {
133  handle->error = (handle->ref - handle->fbk);
134  /* proportional term */
135  handle->outP = _IQmpy(handle->Kp, handle->error);
136 
137  /* integral term */
138  handle->outI += _IQrepeat(handle->Ki, handle->error);
139  handle->outI = _IQsat(handle->outI, handle->outIMax, handle->outIMin);
140 
141  /* control output */
142  handle->preOut = handle->outP + handle->outI;
143  handle->out = _IQsat(handle->preOut , handle->outMax, handle->outMin);
144 }
