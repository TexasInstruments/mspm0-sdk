109 __STATIC_INLINE void SPDEST_UpdateParams(SPDEST_Instance *handle,
110  SPDEST_PARA *para)
111 {
112  handle->K1 = _IQ21(1/(para->baseFreq*para->Ts));
113  handle->K2 = _IQ21(1/(1+para->Ts*2*SPDEST_PI*SPDEST_FILTER_CUTOFF_FREQ));
114  handle->K3 = _IQ21(1)-(handle->K2);
115  handle->BaseRpm = para->baseRpm;
116 }

122 __STATIC_INLINE void SPDEST_run(SPDEST_Instance *handle)
123 {
124  /* Synchronous speed computation */
125  handle->Temp = handle->EstimatedTheta - handle->OldEstimatedTheta;
126  ANGLE_WRAP(&handle->Temp);
127  handle->Temp = _IQ21mpy(handle->K1, _IQtoIQ21(handle->Temp));
128 
129  /* Low-pass filter */
130  /* Q21 = Q21*Q21 + Q21*Q21 */
131  handle->Temp = _IQ21mpy(handle->K2, _IQtoIQ21(handle->EstimatedSpeed))
132  +_IQ21mpy(handle->K3, handle->Temp);
133 
134  /* Saturate the output */
135  handle->Temp=_IQsat(handle->Temp,_IQ21(1),_IQ21(-1));
136  handle->EstimatedSpeed = _IQ21toIQ(handle->Temp);
137 
138  /* Update the electrical angle */
139  handle->OldEstimatedTheta = handle->EstimatedTheta;
140 
141  /* Change motor speed from pu value to rpm value (GLOBAL_Q -> Q0)*/
142  /* Q0 = Q0*GLOBAL_Q => _IQXmpy(), X = GLOBAL_Q*/
143  handle->EstimatedSpeedRpm = _IQmpy(handle->BaseRpm, handle->EstimatedSpeed);
144 }
