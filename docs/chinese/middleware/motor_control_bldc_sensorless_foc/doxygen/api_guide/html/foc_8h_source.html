234 __STATIC_INLINE void FOC_run(FOC_Instance *handle)
235 {
236  if(enableFOC == true)
237  {
238  if(!enablePWM)
239  {
240  enablePWM = true;
241  HAL_PWMEnableChannel(handle->pwmAHal);
242  HAL_PWMEnableChannel(handle->pwmBHal);
243  HAL_PWMEnableChannel(handle->pwmCHal);
244  }
245 /*================================ LEVEL 1 =====================================
246  Level 1 checks the modules, should not connect the motor
247 ==============================================================================*/
248 #if (BUILDLEVEL==LEVEL1)
249  /*---------------------------------------------------------------------
250  Connect inputs of the RMP module and run
251  ----------------------------------------------------------------------*/
252  if(lsw==0)rc.TargetValue = 0;
253  else rc.TargetValue = speedRef;
254  RC_run(&rc);
255  /*----------------------------------------------------------------------
256  Connect inputs of the RAMP GEN module and run
257  ----------------------------------------------------------------------*/
258  rg.Freq = rc.SetpointValue;
259  RG_run(&rg);
260  /*----------------------------------------------------------------------
261  Connect inputs of the IPARK module and call run
262  ----------------------------------------------------------------------*/
263  ipark.Ds = VdTesting;
264  ipark.Qs = VqTesting;
265  ipark.Sine = _IQsinPU(rg.Out);
266  ipark.Cosine = _IQcosPU(rg.Out);
267  IPARK_run(&ipark);
268  /*----------------------------------------------------------------------
269  Connect inputs of the SVGEN module and call run
270  ----------------------------------------------------------------------*/
271  svgen.Ualpha = ipark.Alpha;
272  svgen.Ubeta = ipark.Beta;
273  SVGEN_run(&svgen);
274  /*----------------------------------------------------------------------
275  Connect inputs of the PWMGEN module and call run
276  ----------------------------------------------------------------------*/
277  pwmgen.MfuncC1 = svgen.Ta;
278  pwmgen.MfuncC2 = svgen.Tb;
279  pwmgen.MfuncC3 = svgen.Tc;
280  PWMGEN_run(&pwmgen);
281 
282 #if __ENABLE_LOG
283  FOC_addVarToLog(FOC_LOG_IDX_0, svgen.Ta);
284  FOC_addVarToLog(FOC_LOG_IDX_1, svgen.Tb);
285  FOC_addVarToLog(FOC_LOG_IDX_2, svgen.Tc);
286  FOC_addVarToLog(FOC_LOG_IDX_3, svgen.Tb - svgen.Tc);
287  FOC_addVarToLog(FOC_LOG_IDX_4, rg.Out);
288 #endif
289 #endif /* (BUILDLEVEL==LEVEL1) */
