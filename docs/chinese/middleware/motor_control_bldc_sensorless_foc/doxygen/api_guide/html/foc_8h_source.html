420 __STATIC_INLINE void FOC_runMonitor(FOC_Instance *handle)
421 {
422  bool fault = false;
423 
424  /* check the external fault input */
425 #ifdef USER_DEFAULT_ENABLE_FAULT_INPUT_CHECK
426  if(FAULT_INPUT_STATUS((DRV_HANDLE *)handle->drv_handle))
427  {
428  fault = true;
429  handle->faultStatus.bits.extFaultIn = true;
430  }
431 #endif
432 
433  /* check the fault in phase currents */
434  if(_IQabs(handle->ia) > handle->overCurrentLimit ||
435  _IQabs(handle->ib) > handle->overCurrentLimit ||
436  _IQabs(handle->ia + handle->ib) > handle->overCurrentLimit)
437  {
438  fault = true;
439  handle->faultStatus.bits.ocp = true;
440  }
441 
442  /* check the fault in bus voltage */
443  if(handle->vdc > handle->overVoltageLimit)
444  {
445  fault = true;
446  handle->faultStatus.bits.ovlo = true;
447  }
448 
449  if(handle->vdc < handle->underVoltageLimit)
450  {
451  fault = true;
452  handle->faultStatus.bits.uvlo = true;
453  }
454 
455  if(fault == true)
456  {
457  FOC_setFault(handle);
458  }
459 }

465 __STATIC_INLINE void FOC_CONTROL_run(FOC_Instance *handle)
466 {
467  COMMAND *cmd = &handle->cmd;
468  CONTROL *control = &handle->control;
469 
470  if(handle->cmd.motorState == MOTOR_STATE_STOP &&
471  ((handle->state != MOTOR_STOP) &&
472  (handle->state != MOTOR_IDLE) &&
473  (handle->state != MOTOR_FAULT) &&
474  (handle->state != MOTOR_STANDBY)))
475  {
476  handle->state = MOTOR_STOP;
477  }
478 
479  /* state machine is run after the calibration, also the initial adc sampling
480  * should be complete before the control loop runs*/
481  if(handle->offsetCalib)
482  {
483  switch(handle->state)
484  {
485  case MOTOR_IDLE:
486  handle->enablePWM = 0;
487  TRANSIT_reset(&handle->control.transit);
488  control->align.cnt = 0;
489  control->rampup.rampOut = 0;
490  control->slowFirstCycCmplt = 0;
491  IPD_reset(&ipd);
492  RC_reset(&rc);
493  RG_reset(&rg);
494  CLARKE_reset(&clarke);
495  PARK_reset(&park);
496  IPARK_reset(&ipark);
497  EST_reset(&est);
498  PI_reset(&piSpd);
499  PI_reset(&piId);
500  PI_reset(&piIq);
501  handle->state = MOTOR_STANDBY;
502  break;
503  case MOTOR_STANDBY:
504  if(handle->cmd.motorState == MOTOR_STATE_START)
505  {
506  if(control->startupMethod == FOC_STARTUP_ALIGN)
507  {
508  handle->state = MOTOR_ALIGN;
509  }
510  else if(control->startupMethod == FOC_STARTUP_SLOW_FIRST)
511  {
512  handle->state = MOTOR_SLOW_FIRST_CYC;
513  }
514  else if(control->startupMethod == FOC_STARTUP_IPD)
515  {
516  handle->state = MOTOR_IPD;
517  }
518  /* reset the pwm enable checks */
519  handle->enablePWM = 0;
520  FOC_setPwmEnableStatus(handle, 1);
521  }
522  break;
523  case MOTOR_ALIGN:
524  handle->enablePWM = 1;
525  if(control->align.cnt >= control->align.alignTime)
526  {
527  handle->state = MOTOR_RAMPUP;
528 
529  }
530  else
531  {
532  piIq.ref = control->align.alignCurr;
533  control->align.cnt++;
534  }
535  break;
536  case MOTOR_SLOW_FIRST_CYC:
537  if(control->slowFirstCycCmplt)
538  {
539  handle->state = MOTOR_RAMPUP;
540  }
541  else
542  {
543  handle->enablePWM = 1;
544  rg.Freq = handle->control.slowFirstCycFreq;
545  piIq.ref = control->align.alignCurr;
546  }
547  break;
548  case MOTOR_IPD:
549  if(ipd.ipdState == IPD_CMPLT)
550  {
551  if(handle->cmd.directionReversal)
552  {
553  rg.Angle = ANGLE_MATH_2PIp3_PU - ipd.outAngle;
554  }
555  else
556  {
557  rg.Angle = ipd.outAngle;
558  }
559  handle->state = MOTOR_RAMPUP;
560  }
561  break;
562  case MOTOR_RAMPUP:
563  handle->enablePWM = 1;
564  piIq.ref = control->rampup.rampupCurr;
565  control->rampup.rampOut += control->rampup.rate;
566  if(control->rampup.rampOut >= control->rampup.target)
567  {
568  control->rampup.rampOut = control->rampup.target;
569  if(!handle->cmd.disCL)
570  {
571  handle->state = MOTOR_TRANSIT;
572  rc.SetpointValue = control->rampup.target;
573  piSpd.outI = piIq.fbk>>2;
574  }
575  }
576  break;
577  case MOTOR_TRANSIT:
578  if(control->transit.state == TRANSIT_STATE_COMPLETE)
579  {
580  handle->state = MOTOR_RUN;
581  }
582  break;
583  case MOTOR_RUN:
584  /* This is expected to be empty */
585  break;
586  case MOTOR_STOP:
587  handle->cmd.motorState = MOTOR_STATE_STOP;
588  handle->enablePWM = 0;
589  handle->state = MOTOR_IDLE;
590  break;
591  case MOTOR_FAULT:
592  handle->cmd.motorState = MOTOR_STATE_STOP;
593  handle->enablePWM = 0;
594  if(handle->faultStatus.all == 0)
595  {
596  /* When the user clears the fault the state goes to idle */
597  handle->state = MOTOR_IDLE;
598  }
599  break;
600  default:
601  /* This is expected to be empty */
602  break;
603  }
604 
605  /* Check for fault conditions */
606  FOC_runMonitor(handle);
607  }
608 
609  /* sets the update flag for the user variables */
610  if(++userVarUpdateCounter >= FOC_USERVAR_UPDATE_MS)
611  {
612  userVarUpdateCounter = 0;
613  handle->runUserVarUpdate = 1;
614  }
615 }

621 __STATIC_INLINE void FOC_run(FOC_Instance *handle)
622 {
623  /*----------------------------------------------------------------------
624  Connect inputs of the CLARKE module and call run
625  ----------------------------------------------------------------------*/
626  clarke.As = handle->ia;
627  clarke.Bs = handle->ib;
628  CLARKE_run(&clarke);
629  /*----------------------------------------------------------------------
630  Connect inputs of the PARK module and call run
631  ----------------------------------------------------------------------*/
632  park.Alpha = clarke.Alpha;
633  park.Beta = clarke.Beta;
634  park.Sine = _IQ24toIQ(ANGLE_getSine(park.Angle));
635  park.Cosine = _IQ24toIQ(ANGLE_getSine(park.Angle + ANGLE_MATH_PIp2_PU));
636  PARK_run(&park);
637 
638  if(handle->state == MOTOR_RUN || handle->state == MOTOR_TRANSIT)
639  {
640  if(handle->cmd.outerLoop == FOC_OUTER_LOOP_SPEED)
641  {
642  rc.TargetValue = handle->cmd.speedRef;
643  RC_run(&rc);
644  if (handle->PISpdExecCount >= handle->PISpdExecDivider)
645  {
646  piSpd.ref = rc.SetpointValue;
647  piSpd.fbk = est.freq;
648  /* max output of pi speed is the max measurable current */
649  piSpd.outMax = handle->currentSF>>handle->csaDiv;
650  piSpd.outMin = -piSpd.outMax;
651  piSpd.outIMax = piSpd.outMax;
652  piSpd.outIMin = -piSpd.outIMax;
653  PI_run(&piSpd);
654  piIq.ref = piSpd.out;
655  piId.ref = 0;
656  handle->PISpdExecCount = 1;
657  }
658  else
659  {
660  handle->PISpdExecCount++;
661  }
662  }
663  else
664  {
665  piIq.ref = handle->cmd.IqRef;
666  piId.ref = handle->cmd.IdRef;
667  }
668  }
669 
670  switch (handle->state)
671  {
672  case MOTOR_ALIGN:
673  park.Angle = 0;
674  rg.Angle = 0;
675  break;
676  case MOTOR_SLOW_FIRST_CYC:
677  pAngle = rg.Out;
678  RG_run(&rg);
679  if(rg.Out < pAngle)
680  {
681  handle->control.slowFirstCycCmplt = 1;
682  }
683  park.Angle = rg.Out;
684  break;
685  case MOTOR_IPD:
686  IPD_run(&ipd, &handle->hal,
687  handle->offsetA>>FOC_12_TO_8_BIT_OFFSET_SHIFT,
688  handle->offsetB>>FOC_12_TO_8_BIT_OFFSET_SHIFT,
689  handle->offsetC>>FOC_12_TO_8_BIT_OFFSET_SHIFT);
690  break;
691  case MOTOR_RAMPUP:
692  rg.Freq = handle->control.rampup.rampOut;
693  RG_run(&rg);
694  park.Angle = rg.Out;
695  break;
696  case MOTOR_TRANSIT:
697  rg.Freq = handle->control.rampup.rampOut;
698  RG_run(&rg);
699  park.Angle = TRANSIT_run(&handle->control.transit, est.Theta,
700  rg.Out);
701  break;
702  case MOTOR_RUN:
703  park.Angle = est.Theta;
704  break;
705  default:
706  break;
707  }
708 
709  switch(handle->state)
710  {
711  case MOTOR_IPD:
712  break;
713  case MOTOR_ALIGN:
714  case MOTOR_SLOW_FIRST_CYC:
715  case MOTOR_RAMPUP:
716  case MOTOR_TRANSIT:
717  case MOTOR_RUN:
718  /*------------------------------------------------------------------
719  Connect inputs of the PI ID module and call run
720  ------------------------------------------------------------------*/
721  piId.fbk = park.Ds;
722  piId.outIMax = _IQmpy(handle->vdc, FOC_PIIDQ_LMT_IQ);
723  piId.outIMin = -piId.outIMax;
724  piId.outMax = piId.outIMax;
725  piId.outMin = -piId.outMax;
726  PI_run(&piId);
727  /*------------------------------------------------------------------
728  Connect inputs of the PI IQ module and call run
729  ------------------------------------------------------------------*/
730  piIq.fbk = park.Qs;
731  piIq.outIMax = FOC_getPIIQLMT(handle);
732  piIq.outIMin = -piIq.outIMax;
733  piIq.outMax = piIq.outIMax;
734  piIq.outMin = -piIq.outMax;
735  PI_run(&piIq);
736  /*------------------------------------------------------------------
737  Connect inputs of the PHASEVOLT module and call run
738  ------------------------------------------------------------------*/
739  phaseVolt.DcBusVolt = handle->vdc;
740  phaseVolt.MfuncV1 = svgen.Ta;
741  phaseVolt.MfuncV2 = svgen.Tb;
742  phaseVolt.MfuncV3 = svgen.Tc;
743  PHASEVOLT_run(&phaseVolt);
744  /*------------------------------------------------------------------
745  Connect inputs of the est module and call run
746  ------------------------------------------------------------------*/
747  est.Ialpha = clarke.Alpha;
748  est.Ibeta = clarke.Beta;
749  est.Valpha = phaseVolt.Valpha;
750  est.Vbeta = phaseVolt.Vbeta;
751  EST_run(&est);
752  /*------------------------------------------------------------------
753  Connect inputs of the IPARK module and call run
754  ------------------------------------------------------------------*/
755  ipark.Ds = _IQdiv(piId.out, handle->vdc);
756  ipark.Qs = _IQdiv(piIq.out, handle->vdc);
757  ipark.Sine = park.Sine;
758  ipark.Cosine = park.Cosine;
759  IPARK_run(&ipark);
760  /*------------------------------------------------------------------
761  Connect inputs of the SVGEN module and call run
762  ------------------------------------------------------------------*/
763  svgen.Ualpha = ipark.Alpha;
764  svgen.Ubeta = ipark.Beta;
765  SVGEN_run(&svgen);
766  /*------------------------------------------------------------------
767  Connect inputs of the PWMGEN module and call run
768  ------------------------------------------------------------------*/
769  if(handle->cmd.directionReversal)
770  {
771  pwmgen.MfuncC1 = svgen.Tb;
772  pwmgen.MfuncC2 = svgen.Ta;
773  }
774  else
775  {
776  pwmgen.MfuncC1 = svgen.Ta;
777  pwmgen.MfuncC2 = svgen.Tb;
778  }
779  pwmgen.MfuncC3 = svgen.Tc;
780  PWMGEN_run(&pwmgen);
781 
782  HAL_setPWMCcValue(handle->hal.pwmAHal, pwmgen.dutyA);
783  HAL_setPWMCcValue(handle->hal.pwmBHal, pwmgen.dutyB);
784  HAL_setPWMCcValue(handle->hal.pwmCHal, pwmgen.dutyC);
785 
786  case MOTOR_IDLE:
787  case MOTOR_STOP:
788  case MOTOR_STANDBY:
789  /* pwm control checks to */
790  if(handle->enablePWM)
791  {
792  if(!FOC_getPwmEnableStatus(handle))
793  {
794  /* Enable the PWM channels */
795  HAL_PWMEnableChannel(handle->hal.pwmAHal);
796  HAL_PWMEnableChannel(handle->hal.pwmBHal);
797  HAL_PWMEnableChannel(handle->hal.pwmCHal);
798  FOC_setPwmEnableStatus(handle, 1);
799  }
800  }
801  else
802  {
803  if(FOC_getPwmEnableStatus(handle))
804  {
805  /* Disable the PWM channels */
806  HAL_PWMDisableChannel(handle->hal.pwmAHal);
807  HAL_PWMDisableChannel(handle->hal.pwmBHal);
808  HAL_PWMDisableChannel(handle->hal.pwmCHal);
809  FOC_setPwmEnableStatus(handle, 0);
810  }
811  }
812  break;
813  default:
814  /* This is expected to be empty */
815  break;
816  }
817 }

844 __STATIC_INLINE void FOC_loop(FOC_Instance *handle, uint16_t vdcAdc,
845  uint16_t iaAdc, uint16_t ibAdc, uint16_t icAdc)
846 {
847  _iq iaTemp, ibTemp;
848  FOC_UpdateADCRaw(handle, vdcAdc, iaAdc, ibAdc, icAdc);
849  handle->isrTicker++;
