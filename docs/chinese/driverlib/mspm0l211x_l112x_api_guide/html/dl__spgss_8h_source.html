215 __STATIC_INLINE void DL_SPGSS_setLoopbackConfig(SPGSS_Regs *spgss, DL_SPGSS_PAIR_INDEX index, DL_SPGSS_LoopbackConfig *config)
216 {
217  volatile uint32_t *pReg = &spgss->I2C_PAIR.PAIR0;
218 
219  DL_Common_updateReg((pReg + (uint32_t) index),
220  ((uint32_t) config->controller | (uint32_t) config->target),
221  (SPGSS_PAIR0_CTL_MASK | SPGSS_PAIR0_TARGET_MASK));
222 }

234 {
