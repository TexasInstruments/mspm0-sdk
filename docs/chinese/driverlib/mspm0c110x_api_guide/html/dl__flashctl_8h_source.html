981 __STATIC_INLINE uint32_t DL_FlashCTL_getFlashSectorNumberInBank(
982  FLASHCTL_Regs *flashctl, uint32_t addr)
983 {
984  uint8_t numBanks = DL_FactoryRegion_getNumBanks();
985  uint32_t mainFlashSize = DL_FactoryRegion_getMAINFlashSize();
986  uint32_t sector, bankSectors, sectorInBank;
987 
988  /* Current sector over the whole memory map */
989  sector = DL_FlashCTL_getFlashSectorNumber(flashctl, addr);
990  if (numBanks > 1) {
991  /* Number of sectors per bank, considered max sector
992  * count for a bank-adjusted sector. Assume banks are
993  * evenly distributed */
994  bankSectors = (mainFlashSize / numBanks);
995  /* We will not assume that the maximum number of bank sectors is a
996  * multiple of 2. Which does mean incurring a modulo operation. */
997  sectorInBank = sector % bankSectors;
998  } else {
999  sectorInBank = sector;
1000  }
1001 
1002  return sectorInBank;
1003 }
