666 __STATIC_INLINE void DL_I2CC_startTransfer(UNICOMM_Inst_Regs *unicomm,
667  uint32_t targetAddr, DL_I2CC_DIRECTION direction, uint16_t length)
668 {
669  // Specify target address and read/write mode
670  DL_Common_updateReg(&unicomm->i2cc->TA,
671  ((targetAddr << UNICOMMI2CC_TA_ADDR_OFS) | (uint32_t) direction),
672  (UNICOMMI2CC_TA_ADDR_MASK | UNICOMMI2CC_TA_DIR_MASK));
673 
674  // STOP bit is generated after burst length number of bytes transferred
675  DL_Common_updateReg(&unicomm->i2cc->CTR,
676  (((uint32_t) length << UNICOMMI2CC_CTR_BLEN_OFS) |
677  UNICOMMI2CC_CTR_FRM_START_ENABLE | UNICOMMI2CC_CTR_START_ENABLE |
678  UNICOMMI2CC_CTR_STOP_ENABLE),
679  (UNICOMMI2CC_CTR_BLEN_MASK | UNICOMMI2CC_CTR_FRM_START_MASK |
680  UNICOMMI2CC_CTR_START_MASK | UNICOMMI2CC_CTR_STOP_MASK));
681 }
